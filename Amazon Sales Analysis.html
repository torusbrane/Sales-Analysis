def convert_column_to_float(df, column_name):
    """
    Convert a column in a DataFrame to numeric values, handling non-numeric values by 
    converting them to NaN, and then change the data type to float.

    Parameters:
        df: pandas.DataFrame
            the DataFrame to be used.
        
        column_name:  str
            the column to be converted to float.

    Returns:
        pandas.DataFrame: The DataFrame with the specified column converted to float.
    """
    # Convert column to numeric, setting errors='coerce' to handle non-numeric values by converting them to NaN
    df[column_name] = pd.to_numeric(df[column_name], errors='coerce')
    
    # Drop rows where the specified column is NaN
    df.dropna(subset=[column_name], inplace=True)

    # Convert the column to float (optional, as it's likely already in float format after to_numeric)
    df[column_name] = df[column_name].astype(float)
    
    # Reset the index and drop the old index to avoid adding it as a column
    df.reset_index(drop=True, inplace=True)

    return df

def convert_column_to_float_fillna_with_mean(df, column_name):
    """
    Convert a column in a DataFrame to numeric values, converting non-numeric values to NaN, 
    and then change the data type to float. NaN values are then replaced with 
    the average of the column.

    Parameters:
        df: pandas.DataFrame
            The DataFrame to be used.
        column_name: str
            The name of the column to convert.

    Returns:
        pandas.DataFrame
            The DataFrame with the specified column converted to float and NaN filled with the 
            column's average.
    """
    # Convert column to numeric, setting errors='coerce' to handle non-numeric values by converting them to NaN
    df[column_name] = pd.to_numeric(df[column_name], errors='coerce')

    # Calculate the mean of the column, excluding NaN values
    mean_value = df[column_name].mean()

    # Fill NaN values in the column with the calculated mean
    df[column_name].fillna(value=mean_value, inplace=True)
    
    # Round the column values to 1 decimal place
    df[column_name] = df[column_name].round(1)

    return df

  • The con of deleting the contaminated rows - if there is not enough data, it can give less accurate results by deleting the rows.
  • The pro of filling with average - helps to fill in the missing data to keep deleted data to a minimum, which allows for closer approximation of the missing data. This can help when modelling the data, which can help spot things like outliers.
  • The con of filling with average - may add some accidental bias into the dataset that was not there in the beginning.

def convert_column_to_int(df, column_name):
    '''
    Convert the column from the dataframe to pure numeric data, any non-numeric data will be converted to
    NaN, the NaN's will be deleted, and the remaining numeric data will be converted to integers. All
    commas will be deleted to prevent any deletion of numbers higher than 999. (Ex: 1,000)
    
    Parameters:
        df: pandas.DataFrame
            The DataFrame to be used.

        column_name: str
            The column to be converted to int.
            
    Returns:
        pandas.DataFrame
            The DataFrame with the specified column converted to int.
    '''
    # Convert the column to string, remove the commas, convert column to numeric, 
    # setting errors='coerce' to handle non-numeric values by converting them to NaN
    df[column_name] = df[column_name].astype(str).str.replace(',', '').apply(pd.to_numeric, errors='coerce')
        
    # Drop rows where the specified column is NaN
    df.dropna(subset=[column_name], inplace=True)

    # Convert the column to int
    df[column_name] = df[column_name].astype(int)
    
    # Reset the index and drop the old index to avoid adding it as a column
    df.reset_index(drop=True, inplace=True)

    return df

def convert_column_to_int_remove_currency_symbols(df, column_name):
    '''
    Convert the column from the dataframe to pure numeric data, any non-numeric data will be converted to
    NaN, the NaN's will be deleted, and the remaining numeric data will be converted to integers. All
    commas and leading currency symbols will be deleted.
    
    Parameters:
        df: pandas.DataFrame
            The DataFrame to be used.

        column_name: str
            The column to be converted to int.
            
    Returns:
        pandas.DataFrame
            The DataFrame with the specified column converted to int.
    '''
    # Convert the column to string, remove the currency symbols and commas, convert column to numeric, 
    # setting errors='coerce' to handle non-numeric values by converting them to NaN
    df[column_name] = df[column_name].astype(str).str.replace(r'^\D+', '', regex=True).str.replace(',', '')
    
    df[column_name]= pd.to_numeric(df[column_name], errors='coerce')
        
    # Drop rows where the specified column is NaN
    df.dropna(subset=[column_name], inplace=True)

    # Convert the column to int
    df[column_name] = df[column_name].astype(int)
    
    # Reset the index and drop the old index to avoid adding it as a column
    df.reset_index(drop=True, inplace=True)

    return df

def convert_column_to_int_remove_percent(df, column_name):
    '''
    Convert the column from the dataframe to pure numeric data, any non-numeric data will be converted to
    NaN, the NaN's will be deleted, and the remaining numeric data will be converted to integers. All
    commas and trailing symbols will be deleted.
    
    Parameters:
        df: pandas.DataFrame
            The DataFrame to be used.

        column_name: str
            The column to be converted to int.
            
    Returns:
        pandas.DataFrame
            The DataFrame with the specified column converted to int.
    '''
    # Convert the column to string, remove the percent sign and commas, convert column to numeric, 
    # setting errors='coerce' to handle non-numeric values by converting them to NaN
    df[column_name] = df[column_name].astype(str).str.replace(r'\D+$', '', regex=True).str.replace(',', '')
    
    df[column_name]= pd.to_numeric(df[column_name], errors='coerce')
        
    # Drop rows where the specified column is NaN
    df.dropna(subset=[column_name], inplace=True)

    # Convert the column to int
    df[column_name] = df[column_name].astype(int)
    
    # Reset the index and drop the old index to avoid adding it as a column
    df.reset_index(drop=True, inplace=True)

    return df

def convert_object_columns_to_string(df):
    """
    Convert a column in a DataFrame to strings.

    Parameters:
        df: pandas.DataFrame
            The DataFrame to be used.

    Returns:
        pandas.DataFrame
            The DataFrame with the specified column converted to string from object dtype.
    """
    # Find columns of dtype 'object'
    object_columns = df.select_dtypes(include='object').columns
    
    # Convert each object columns to string type
    for column_name in object_columns:
        df[column_name] = df[column_name].astype(str)

    return df

Since the category titles are very long, and Amazon shows every category and subcategory added together, we can extract just the last word in the category title. So the most occuring category is "USBCables". Let's write a function that takes care of that for us¶

In [71]:
def extract_last_word(categories):
    # This list will store the last words for each category
    last_words = []
    
    # Iterate over each category in the input list
    for category in categories:
        # Split the category by '|' and take the last element
        last_word = category.split('|')[-1]
        # Add the last word to the list
        last_words.append(last_word)
    
    return last_words
